#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[n][n];
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
            cin>>a[i][j];
    for(int i=0;i<n;i++)
        for(int j=0;j<n-1;j++)
            for(int k=j+1;k<n;k++)
                if(a[i][j]>a[i][k])
                    swap(a[i][j],a[i][k]);
    for(int i=0;i<n;i++)
        for(int j=i+1;j<n;j++)
            swap(a[i][j],a[j][i]);
    for(int i=0;i<n;i++)
        for(int j=0;j<n-1;j++)
            for(int k=j+1;k<n;k++)
                if(a[i][j]>a[i][k])
                    swap(a[i][j],a[i][k]);
    for(int i=0;i<n;i++)
        for(int j=i+1;j<n;j++)
            swap(a[i][j],a[j][i]);
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++)
            cout<<a[i][j]<<" ";
        cout<<'\n';
    }
    return 0;
}
